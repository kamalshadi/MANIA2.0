p = []
S1 = {}
S2 = {}
for n in N:
    S1[n] = {}
    env1 = [(xx['envelope'],xx['n2'],xx['slope']) for xx in A if (xx['n1']==n)]
    tit1 = 'Source '+n
    S2[n] = {}
    env2 = [(xx['envelope'],xx['n1'],xx['slope']) for xx in A if (xx['n2']==n)]
    tit2 = 'Target '+n
    s1 = figure(plot_width=400, plot_height=400, title = tit1, y_axis_label = "Probability(log scale)",x_axis_label = "Distance")
    s2 = figure(plot_width=400, plot_height=400, title = tit2, y_axis_label = "Probability(log scale)",x_axis_label = "Distance")
    for target in env1:
        tar = target[0]
        S1[n][target[1]] = target[2]
        try:
            color = C[target[1]]
        except KeyError:
            color = gen_hex_colour_code()
            C[target[1]] = color
#     flat_list = [item for sub in env for item in sub]
        x = [xx[0] for xx in tar if xx[0]>1]
        y = [xx[1] for xx in tar if xx[0]>1]
#     tit = 'Source '+n
        s1.circle(x,y,color=color)
    for source in env2:
        src = source[0]
        S2[n][source[1]] = source[2]
        try:
            color = C[source[1]]
        except KeyError:
            color = gen_hex_colour_code()
            C[source[1]] = color
#     flat_list = [item for sub in env for item in sub]
        x = [xx[0] for xx in src if xx[0]>1]
        y = [xx[1] for xx in src if xx[0]>1]
#     tit = 'Source '+n
        s2.circle(x,y,color=color)
    p.append([s1,s2])

color = ['red','blue','green','black']

rs = [0]*4
rt = [0]*4
txt = ['a']*4
tt = ['a']*4
s = figure(plot_width=600, plot_height=400, y_axis_label = "Slopes to targets/Sources",x_axis_label = "Target/Source")
for i,n1 in enumerate(S1.keys()):
    txt[i] = "Source "+n1
    val = [-S1[n1][xx]/100.0 for xx in S1[n1]]
    rs[i] = s.circle(2*i,val,color=color[i])
for i,n1 in enumerate(S2.keys()):
    tt[i] = "target "+n1
    val = [-S2[n1][xx]/100.0 for xx in S2[n1]]
    rt[i] = s.square(2*i+1,val,color=color[i])
legend = Legend(items=[(txt[i],[rs[i]]) for i in range(4)]+[(tt[i],[rt[i]]) for i in range(4)], location=(1, 30))
s.add_layout(legend, 'right')

show(s)
